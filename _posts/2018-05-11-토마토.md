# 토마토

https://www.acmicpc.net/problem/7576



```java
package com.company;

import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class Main {

    static int [][] box;
    static int dx[] = {0,0,1,-1};
    static int dy[] = {1,-1,0,0};
    static int N,M;
    static Queue<Tomato> q = new LinkedList<Tomato>();

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        M = sc.nextInt(); //가로
        N = sc.nextInt(); //세로
        box = new int [N][M]; //세로 가로
        boolean checker = true;
        //1 익은 토마토 / 0 익지 않은 토마토 / -1 토마토 없음

        for (int i =0;i<N;i++){
            for (int j =0;j<M;j++){

                box [i][j] = sc.nextInt(); //박스에 담음.

                //익은 토마토인 경우
                if(box[i][j] == 1){
                    q.add(new Tomato(i,j,0));

                } else { //익지 않은 토마토이거나 없을 경우
                    checker = false;
                }
            }

            if (checker) { //처음부터 저장될 때 모든 토마토가 익은 경우
                System.out.println("0");
                return ;
            }
            bfs();
            //checker = true;
            int days = 0;

            for(int a=0;a<N;a++){
                for(int b=0;b<M;b++){

                    if(box[a][b] > days){
                        days = box[a][b];
                    }
                    if (box[a][b] == 0){ //토마토가 모두 익지 못하는 상황일 시
                        System.out.println("-1");
                        return ;
                    }
                }
            }
            System.out.println(days); //최소 날짜 출력
        }

    }

    static void bfs(){
        while (!q.isEmpty()) {
            Tomato p = q.remove();

            for(int i = 0; i<4;i++){
                int cx = p.x+dx[i];
                int cy = p.y+dy[i];

                if(cx >= 0 && cx < N && cy >= 0 && cy < M && box[cx][cy]==0){
                    q.add(new Tomato(cx,cy,p.count+1));
                    box[cx][cy]=p.count+1;
                }
            }
        }
    }
}

class Tomato {
    int x,y,count;
    Tomato(int x, int y, int count){
        this.x = x;
        this.y = y;
        this.count = count;
    }
}
```
